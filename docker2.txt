Docker - это платформа для разработчиков, позволяет запускать приложение в контейнерах 
Контейнеры представляет собой изолированные среды, в которых запускаются приложения и их зависимости

Image(оброзы) -

Основные команды
docker images - показывает установленые образы
docker run -d -p 8080:80 nginx создание контейнерах
docker ps - просмотр
docker ps -a - просмотр всех
docker rmi "name_images" - удаление образов
docker rm "conteiner_id" - удаление контейнера 
docker stop "conteiner_id" - остановка
docker start  "conteiner_id"- старт
docker rm $(docker ps -aq) - удаление всех контейнеров
docker kill "conteiner_id" - удаляет сразу контейнер
docker exec -it "conteiner_id" bash - запускает терминал запущенного контейнера
docker pull "name_images" - установление образа

Dockerfile:
FROM - главный образ, с которого идет сборка
MAINTAINER name <mail> - кто собрал образ 
RUN - запускает команды
CMD - запускает процессы 
COPY - копирует фалйы из нашего хоста во внутрь контейнера ( /usr/src/app) - первый указ хост, кторым контейнер
EXPOSE - открыть просмотр
WORKDIR - устанавливает рабочую директорию, мы попадаем в нее при заходе внутрь контейнера
